WITH wallet_balances AS (
   -- debits
   SELECT array_to_string(inputs.addresses, ",") AS address, -inputs.value AS value
   FROM `bigquery-public-data.crypto_bitcoin.inputs` AS inputs
   UNION ALL
   -- credits
   SELECT array_to_string(outputs.addresses, ",") AS address, outputs.value AS value
   FROM `bigquery-public-data.crypto_bitcoin.outputs` AS outputs
),
top_addresses AS (
   SELECT address, SUM(value) AS balance
   FROM wallet_balances
   GROUP BY address
   ORDER BY balance DESC
   LIMIT 100
),
total_balance AS (
   SELECT SUM(value) AS total
   FROM wallet_balances
)
SELECT address, 
       balance,
       ROUND(balance / (SELECT total FROM total_balance) * 100, 9) AS top_100_holdings_percent
FROM top_addresses;
