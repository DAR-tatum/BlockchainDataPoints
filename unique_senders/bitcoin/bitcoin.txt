WITH recievers AS ( 
   SELECT 
      FORMAT_TIMESTAMP('%Y-%m-%d', block_timestamp) AS date,
      array_to_string(inputs.addresses, ",") AS address,
   FROM `bigquery-public-data.crypto_bitcoin.inputs` AS inputs
)
SELECT date, COUNT(DISTINCT address) AS unique_senders
FROM recievers
GROUP BY date
ORDER BY date DESC;
