SELECT 
    EXTRACT(YEAR FROM block_timestamp) AS year,
    EXTRACT(MONTH FROM block_timestamp) AS month,
  COUNT(*) AS failed_transaction_count,
  from_address AS sender,
  to_address AS receiver
FROM `bigquery-public-data.crypto_ethereum.transactions`
WHERE receipt_status = 0
GROUP BY year, month, sender, receiver
ORDER BY failed_transaction_count DESC;
