WITH wallets AS (
   SELECT 
      FORMAT_TIMESTAMP('%Y-%m-%d', block_timestamp) AS date,
      array_to_string(inputs.addresses, ",") AS address
   FROM `bigquery-public-data.crypto_zcash.inputs` AS inputs
   UNION ALL
   SELECT 
      FORMAT_TIMESTAMP('%Y-%m-%d', block_timestamp) AS date,
      array_to_string(outputs.addresses, ",") AS address
   FROM `bigquery-public-data.crypto_zcash.outputs` AS outputs
)
SELECT 
  date,
  COUNT(DISTINCT address) AS total_num_addresses
FROM wallets
GROUP BY date
ORDER BY date DESC;
